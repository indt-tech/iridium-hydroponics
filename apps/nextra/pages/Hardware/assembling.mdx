import React, {useState} from 'react';

export function Dialog(props) {
  const [isDialogOpen, setDialogOpen] = useState(false)

  const toggleDialog = () => {
    setDialogOpen(!isDialogOpen)
  }
  if (!isDialogOpen) return (<div style={{cursor:"zoom-in"}} onClick={toggleDialog}>{props.children}</div>)
  return (
    <div style={{ position: 'fixed', top: 0, left: 0, right: 0, bottom: 0, backgroundColor: 'rgba(0,0,0,0.8)', display:'flex', alignItems:'center', justifyContent:'center' }} onClick={toggleDialog}>
      <div style={{ maxWidth: '110vh', backgroundColor: 'white', padding: '20px', width:'80vw', height:'80vh' }} onClick={e => e.stopPropagation()}>
        {props.children}
      </div>
    </div>
  )
}
